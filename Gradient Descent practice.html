sse_all = []
slopes = []
intercepts = []
for m_old in np.linspace(-1,1,200):
    for c_old in np.linspace(-10, 10, 200):
        #m_old = 0.5
        #c_old = 2
    
        yhat = m_old * adv['TV'] + c_old
        error = adv['sales'] - yhat
        sq_error = error.apply(lambda v: np.square(v))
        sse = 1/adv.shape[0] * sq_error.sum()
        sse_all.append(sse)
        slopes.append(m_old)
        intercepts.append(c_old)
